from nada_dsl import *
def nada_main():
data_owner = Party(name="data_owner")
start = SecretInteger(Input(name="start", party=data_owner))
sequence: list[SecretInteger] = []
for i in range(3):
sequence.append(start + Integer(i))
outputs: list[Output] = []
for i in range(3):
outputs.append(Output(
sequence[i],
"sequence_" + str(i),
data_owner
))
return outputs
